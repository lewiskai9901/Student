## 🚨 风险评估与预防

### 已知风险点及解决方案

| 风险 | 影响 | 概率 | 预防措施 | 回滚方案 |
|------|------|------|----------|----------|
| 数据库迁移失败 | 高 | 低 | 先在测试环境执行，备份数据 | `git revert` + 数据恢复脚本 |
| 后端编译失败 | 中 | 中 | 每次改动后立即编译验证 | `git checkout -- <file>` |
| 前端类型错误 | 中 | 中 | TypeScript strict模式检查 | `git checkout` + `npm run build` |
| API不兼容导致前端报错 | 高 | 低 | 保持API向后兼容 | 使用feature flag控制 |
| 性能下降 | 中 | 低 | 添加数据库索引，缓存优化 | 监控性能指标 |
| 数据丢失 | 高 | 极低 | 每步操作前数据库备份 | 从备份恢复 |

### 1.2 后端 - buildTreeNode方法改造（3小时）

#### 📁 文件：`UniversalPlaceApplicationService.java`

**改动位置**：`buildTreeNode` 方法（约384-440行）

**改动内容**：

```java
// 改动前
private PlaceTreeNode buildTreeNode(UniversalPlace place) {
    PlaceTreeNode node = new PlaceTreeNode();
    // ... 设置基础字段
    node.setOrgUnitId(place.getOrgUnitId());
    // 查询组织名称
    if (place.getOrgUnitId() != null) {
        orgUnitRepository.findById(place.getOrgUnitId())
                .ifPresent(org -> node.setOrgUnitName(org.getUnitName()));
    }
    // ... 其他逻辑
}

// 改动后
private PlaceTreeNode buildTreeNode(UniversalPlace place) {
    PlaceTreeNode node = new PlaceTreeNode();
    // ... 设置基础字段

    // 🔄 改造：组织信息 + 继承判断
    node.setOrgUnitId(place.getOrgUnitId());

    // 获取父场所（用于继承判断）
    UniversalPlace parent = null;
    if (place.getParentId() != null) {
        parent = placeRepository.findById(place.getParentId()).orElse(null);
    }

    // 设置父组织ID
    if (parent != null) {
        node.setParentOrgUnitId(parent.getOrgUnitId());
    }

    // 判断是否继承（当前组织 == 父组织 && 当前组织不为空）
    boolean isInherited = false;
    if (place.getOrgUnitId() != null && parent != null && parent.getOrgUnitId() != null) {
        isInherited = place.getOrgUnitId().equals(parent.getOrgUnitId());
    }
    node.setIsOrgInherited(isInherited);

    // 查询组织名称
    if (place.getOrgUnitId() != null) {
        orgUnitRepository.findById(place.getOrgUnitId())
                .ifPresent(org -> node.setOrgUnitName(org.getUnitName()));
    }

    // 🔄 改造：负责人信息 + 继承判断（可选）
    node.setResponsibleUserId(place.getResponsibleUserId());

    if (parent != null) {
        node.setParentResponsibleUserId(parent.getResponsibleUserId());
    }

    boolean isResponsibleInherited = false;
    if (place.getResponsibleUserId() != null && parent != null && parent.getResponsibleUserId() != null) {
        isResponsibleInherited = place.getResponsibleUserId().equals(parent.getResponsibleUserId());
    }
    node.setIsResponsibleInherited(isResponsibleInherited);

    // 查询负责人名称
    if (place.getResponsibleUserId() != null) {
        userRepository.findById(place.getResponsibleUserId())
                .ifPresent(user -> node.setResponsibleUserName(user.getRealName()));
    }

    // ... 其他逻辑保持不变
}
```

```java
package com.school.management.infrastructure.event;

import com.school.management.domain.place.model.aggregate.UniversalPlace;
import com.school.management.infrastructure.persistence.place.PlaceChangeHistoryMapper;
import com.school.management.infrastructure.persistence.place.PlaceChangeHistoryPO;
import lombok.RequiredArgsConstructor;
import lombok.extern.slf4j.Slf4j;
import org.springframework.context.event.EventListener;
import org.springframework.stereotype.Component;
import org.springframework.transaction.event.TransactionalEventListener;

import java.time.LocalDateTime;

/**
 * 场所变更事件监听器
 * 自动记录场所关键字段变更到历史表
 */
@Slf4j
@Component
@RequiredArgsConstructor
public class PlaceChangeEventListener {

    private final PlaceChangeHistoryMapper historyMapper;

    /**
     * 监听场所更新事件
     * 在事务提交后执行，确保主数据已保存
     */
    @TransactionalEventListener
    public void onPlaceUpdated(PlaceUpdatedEvent event) {
        try {
            UniversalPlace oldPlace = event.getOldPlace();
            UniversalPlace newPlace = event.getNewPlace();
            Long operatorId = event.getOperatorId();

            // 检查组织变更
            if (!equals(oldPlace.getOrgUnitId(), newPlace.getOrgUnitId())) {
                recordChange(
                    newPlace.getId(),
                    "ORG_TRANSFER",
                    "org_unit_id",
                    String.valueOf(oldPlace.getOrgUnitId()),
                    String.valueOf(newPlace.getOrgUnitId()),
                    operatorId,
                    "组织归属变更"
                );
            }

            // 检查容量变更
            if (!equals(oldPlace.getCapacity(), newPlace.getCapacity())) {
                recordChange(
                    newPlace.getId(),
                    "CAPACITY_CHANGE",
                    "capacity",
                    String.valueOf(oldPlace.getCapacity()),
                    String.valueOf(newPlace.getCapacity()),
                    operatorId,
                    "容量调整"
                );
            }

            // 检查状态变更
            if (!equals(oldPlace.getStatus(), newPlace.getStatus())) {
                recordChange(
                    newPlace.getId(),
                    "STATUS_CHANGE",
                    "status",
                    oldPlace.getStatus().name(),
                    newPlace.getStatus().name(),
                    operatorId,
                    "状态变更"
                );
            }

            // 检查负责人变更
            if (!equals(oldPlace.getResponsibleUserId(), newPlace.getResponsibleUserId())) {
                recordChange(
                    newPlace.getId(),
                    "RESPONSIBLE_CHANGE",
                    "responsible_user_id",
                    String.valueOf(oldPlace.getResponsibleUserId()),
                    String.valueOf(newPlace.getResponsibleUserId()),
                    operatorId,
                    "负责人变更"
                );
            }

        } catch (Exception e) {
            log.error("记录场所变更历史失败", e);
            // 不抛出异常，避免影响主流程
        }
    }

    /**
     * 记录单个变更
     */
    private void recordChange(Long placeId, String changeType, String fieldName,
                             String oldValue, String newValue, Long changedBy, String reason) {
        PlaceChangeHistoryPO history = new PlaceChangeHistoryPO();
        history.setPlaceId(placeId);
        history.setChangeType(changeType);
        history.setFieldName(fieldName);
        history.setOldValue(oldValue);
        history.setNewValue(newValue);
        history.setChangedBy(changedBy);
        history.setChangedAt(LocalDateTime.now());
        history.setReason(reason);

        historyMapper.insert(history);

        log.info("场所变更记录：placeId={}, type={}, {}={}->{}",
                 placeId, changeType, fieldName, oldValue, newValue);
    }

    /**
     * 安全的对象相等判断
     */
    private boolean equals(Object a, Object b) {
        if (a == null && b == null) return true;
        if (a == null || b == null) return false;
        return a.equals(b);
    }
}

    private UniversalPlace newPlace;
    private Long operatorId;
```

```java
/**
 * 获取场所变更历史
 */
public List<PlaceChangeHistoryDTO> getChangeHistory(Long placeId, String changeType,
                                                     LocalDateTime startTime, LocalDateTime endTime,
                                                     int pageNum, int pageSize) {
    // 构建查询条件
    QueryWrapper<PlaceChangeHistoryPO> query = new QueryWrapper<>();
    query.eq("place_id", placeId);

    if (changeType != null && !changeType.isBlank()) {
        query.eq("change_type", changeType);
    }
    if (startTime != null) {
        query.ge("changed_at", startTime);
    }
    if (endTime != null) {
        query.le("changed_at", endTime);
    }

    query.orderByDesc("changed_at");

    // 分页查询
    Page<PlaceChangeHistoryPO> page = new Page<>(pageNum, pageSize);
    Page<PlaceChangeHistoryPO> result = historyMapper.selectPage(page, query);

    // 转换为DTO
    return result.getRecords().stream()
            .map(this::toChangeHistoryDTO)
            .collect(Collectors.toList());
}

private PlaceChangeHistoryDTO toChangeHistoryDTO(PlaceChangeHistoryPO po) {
    PlaceChangeHistoryDTO dto = new PlaceChangeHistoryDTO();
    dto.setId(po.getId());
    dto.setPlaceId(po.getPlaceId());
    dto.setChangeType(po.getChangeType());
    dto.setFieldName(po.getFieldName());
    dto.setOldValue(po.getOldValue());
    dto.setNewValue(po.getNewValue());
    dto.setChangedBy(po.getChangedBy());
    dto.setChangedAt(po.getChangedAt());
    dto.setReason(po.getReason());

    // 查询操作人姓名
    if (po.getChangedBy() != null) {
        userRepository.findById(po.getChangedBy())
                .ifPresent(user -> dto.setChangedByName(user.getRealName()));
    }

    return dto;
}
```

```java
    BatchOperationResult result = new BatchOperationResult();

    for (Long placeId : placeIds) {
        try {
            UniversalPlace place = placeRepository.findById(placeId)
                    .orElseThrow(() -> new IllegalArgumentException("场所不存在: " + placeId));

            switch (status) {

                    place.enable();
                    break;

                    place.disable();
                    break;

            result.addFailure(placeId, e.getMessage());
        }
    }

    return result;
```

```java
package com.school.management.domain.place.service;

import com.school.management.domain.place.model.aggregate.UniversalPlace;
import com.school.management.domain.place.repository.UniversalPlaceRepository;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;

import java.util.ArrayList;
import java.util.List;

@Service
@RequiredArgsConstructor
public class PlaceCapacityService {

    private final UniversalPlaceRepository placeRepository;

    private static final double WARNING_THRESHOLD = 0.8;   // 80%
    private static final double CRITICAL_THRESHOLD = 0.95; // 95%

    public List<CapacityAlert> getCapacityAlerts() {
        List<UniversalPlace> allPlaces = placeRepository.findAll();
        List<CapacityAlert> alerts = new ArrayList<>();

        for (UniversalPlace place : allPlaces) {
            if (place.getCapacity() == null || place.getCapacity() == 0) {
                continue;
            }

            int capacity = place.getCapacity();
            int occupancy = place.getCurrentOccupancy() != null ? place.getCurrentOccupancy() : 0;
            double rate = (double) occupancy / capacity;

            if (rate >= CRITICAL_THRESHOLD) {
                alerts.add(new CapacityAlert(
                    place.getId(),
                    place.getPlaceName(),
                    capacity,
                    occupancy,
                    rate,
                    AlertLevel.CRITICAL,
                    "容量严重不足，使用率已达 " + String.format("%.1f%%", rate * 100)
                ));
            } else if (rate >= WARNING_THRESHOLD) {
                alerts.add(new CapacityAlert(
                    place.getId(),
                    place.getPlaceName(),
                    capacity,
                    occupancy,
                    rate,
                    AlertLevel.WARNING,
                    "容量接近上限，使用率已达 " + String.format("%.1f%%", rate * 100)
                ));
            }
        }

        alerts.sort((a, b) -> Double.compare(b.getOccupancyRate(), a.getOccupancyRate()));
        return alerts;
    }

    @lombok.Data
    @lombok.AllArgsConstructor
    public static class CapacityAlert {
        private Long placeId;
        private String placeName;
        private int capacity;
        private int occupancy;
        private double occupancyRate;
        private AlertLevel level;
        private String message;
    }

    public enum AlertLevel {
        WARNING, CRITICAL
    }
}
```
