## 📋 方案说明

本方案整合了以下设计：
- ✅ v2.0 高级设计方案（业界最佳实践参考）
- ✅ v2.1 详细实施方案（逐步实施指南）
- ✅ 架构缺陷分析（对比大厂，修复5个P0严重缺陷）

**核心改进**：
1. **继承模型** - 从值拷贝改为NULL继承（对标Kubernetes ConfigMap）
2. **变更审计** - 从简单记录改为完整审计日志（对标AWS CloudTrail）
3. **批量操作** - 从同步阻塞改为异步Job模式（对标AWS Batch Operations）
4. **容量管理** - 从全表扫描改为索引+缓存（对标AWS CloudWatch）
5. **并发控制** - 从无控制改为数据库原子操作（对标Google Spanner）
6. **领域事件** - 引入事件驱动架构（符合DDD最佳实践）

## 🎯 核心改动详解

### 改动1：继承模型重构（NULL继承）

#### 数据库迁移

```sql
-- V23.0.0__place_inheritance_refactor.sql

-- 1. 允许org_unit_id为NULL（表示继承）
ALTER TABLE places MODIFY COLUMN org_unit_id BIGINT NULL;
ALTER TABLE places MODIFY COLUMN responsible_user_id BIGINT NULL;

-- 2. 迁移现有数据（可选：将完全继承父级的数据改为NULL）
UPDATE places p
JOIN places parent ON p.parent_id = parent.id
SET p.org_unit_id = NULL
WHERE p.org_unit_id = parent.org_unit_id
  AND p.parent_id IS NOT NULL;

-- 3. 添加函数：计算有效组织ID（MySQL不支持递归，使用存储过程）
DELIMITER $$
CREATE FUNCTION get_effective_org_unit_id(place_id BIGINT)
RETURNS BIGINT
DETERMINISTIC
BEGIN
  DECLARE org_id BIGINT;
  DECLARE parent_id BIGINT;
  DECLARE max_depth INT DEFAULT 10;
  DECLARE depth INT DEFAULT 0;

  SELECT org_unit_id, parent_id INTO org_id, parent_id
  FROM places WHERE id = place_id;

  -- 如果有明确指定，直接返回
  IF org_id IS NOT NULL THEN
    RETURN org_id;
  END IF;

  -- 递归查找父级（最多10层）
  WHILE parent_id IS NOT NULL AND depth < max_depth DO
    SELECT org_unit_id, parent_id INTO org_id, parent_id
    FROM places WHERE id = parent_id;

    IF org_id IS NOT NULL THEN
      RETURN org_id;
    END IF;

    SET depth = depth + 1;
  END WHILE;

  RETURN NULL;
END$$
DELIMITER ;
```

#### 领域模型

```java
// domain/place/model/aggregate/UniversalPlace.java

public class UniversalPlace extends AggregateRoot {

    private Long id;
    private Long parentId;
    private Long orgUnitId;  // NULL = 继承父级
    private Long responsibleUserId;  // NULL = 继承父级

    /**
     * 获取有效的组织ID（实时计算）
     */
    public Long getEffectiveOrgUnitId() {
        if (this.orgUnitId != null) {
            return this.orgUnitId;  // 明确指定
        }

        // 递归查找父级
        if (this.parent != null) {
            return this.parent.getEffectiveOrgUnitId();
        }

        return null;
    }

    /**
     * 是否继承组织
     */
    public boolean isOrgInherited() {
        return this.orgUnitId == null && this.parentId != null;
    }

    /**
     * 分配组织（覆盖继承）
     */
    public void assignOrganization(Long orgUnitId, OrgUnit orgUnit) {
        // 业务规则验证
        if (orgUnit == null || !orgUnit.isEnabled()) {
            throw new DomainException("组织不存在或已禁用");
        }

        if (Objects.equals(this.orgUnitId, orgUnitId)) {
            return;  // 无变更
        }

        Long oldOrgId = this.getEffectiveOrgUnitId();

        // 更新状态
        this.orgUnitId = orgUnitId;

        // 发布领域事件
        this.registerEvent(new PlaceOrgAssignedEvent(
            this.id,
            this.placeName,
            oldOrgId,
            orgUnitId,
            getCurrentUserId(),
            LocalDateTime.now()
        ));
    }

    /**
     * 恢复组织继承
     */
    public void clearOrgOverride() {
        if (this.orgUnitId == null) {
            return;  // 已经是继承状态
        }

        if (this.parentId == null) {
            throw new DomainException("根场所不能恢复继承");
        }

        Long oldEffectiveOrgId = this.orgUnitId;
        this.orgUnitId = null;

        // 发布事件
        this.registerEvent(new PlaceOrgInheritanceRestoredEvent(
            this.id,
            oldEffectiveOrgId,
            this.getEffectiveOrgUnitId()
        ));
    }
}
```

```java
// application/place/UniversalPlaceApplicationService.java

@Service
@Transactional
public class UniversalPlaceApplicationService {

    /**
     * 分配组织（调用领域方法）
     */
    public void assignOrganization(Long placeId, Long orgUnitId) {
        // 1. 加载聚合根
        UniversalPlace place = placeRepository.findById(placeId).orElseThrow();
        OrgUnit orgUnit = orgUnitRepository.findById(orgUnitId).orElseThrow();

        // 2. 调用领域方法
        place.assignOrganization(orgUnitId, orgUnit);

        // 3. 保存（事件自动发布）
        placeRepository.save(place);
    }

    /**
     * 恢复继承
     */
    public void clearOrgOverride(Long placeId) {
        UniversalPlace place = placeRepository.findById(placeId).orElseThrow();
        place.clearOrgOverride();
        placeRepository.save(place);
    }

    /**
     * 构建树节点（支持继承标识）
     */
    private PlaceTreeNode buildTreeNode(UniversalPlace place) {
        PlaceTreeNode node = new PlaceTreeNode();

        // 基础字段
        node.setId(place.getId());
        node.setPlaceName(place.getPlaceName());

        // 组织信息（继承感知）
        node.setOrgUnitId(place.getOrgUnitId());  // 可能为NULL
        node.setEffectiveOrgUnitId(place.getEffectiveOrgUnitId());  // 计算后的值
        node.setIsOrgInherited(place.isOrgInherited());

        if (place.getParent() != null) {
            node.setParentOrgUnitId(place.getParent().getEffectiveOrgUnitId());
        }

        // 查询组织名称
        if (place.getEffectiveOrgUnitId() != null) {
            orgUnitRepository.findById(place.getEffectiveOrgUnitId())
                .ifPresent(org -> node.setOrgUnitName(org.getUnitName()));
        }

        // ... 其他字段

        return node;
    }
}
```

```java
            .userName(ctx.getUserName())
            .userType(ctx.getUserType())
            .accessKeyId(ctx.getJwtTokenId())
            .sessionId(ctx.getSessionId())
            .mfaAuthenticated(ctx.isMfaAuthenticated())

            // 请求上下文
            .sourceIp(ctx.getSourceIp())
            .userAgent(ctx.getUserAgent())
            .referer(ctx.getReferer())
            .apiEndpoint(ctx.getRequestUri())
            .httpMethod(ctx.getHttpMethod())

            // 变更内容
            .changedFields(buildChangedFields(event))

            // 元数据
            .reason("组织分配变更")
            .ttlExpiresAt(LocalDateTime.now().plusYears(7))

            .build();

        auditLogMapper.insert(log);
        log.info("审计日志已记录: eventId={}, userId={}, placeId={}",
                 log.getEventId(), log.getUserId(), event.getPlaceId());
    }

    private List<FieldChange> buildChangedFields(PlaceOrgAssignedEvent event) {
        return List.of(new FieldChange(
            "orgUnitId",
            event.getOldOrgId(),
            event.getNewOrgId()
        ));
    }
```

```java
// application/place/PlaceBatchJobService.java

@Service
@RequiredArgsConstructor
public class PlaceBatchJobService {

    private final PlaceBatchJobRepository jobRepository;
    private final PlaceBatchJobExecutor jobExecutor;

    /**
     * 提交批量分配组织Job（立即返回）
     */
    public BatchJobDTO submitBatchAssignOrgJob(List<Long> placeIds, Long orgUnitId) {
        // 1. 创建Job记录
        PlaceBatchJob job = PlaceBatchJob.create(
            "BATCH_ASSIGN_ORG",
            placeIds,
            Map.of("orgUnitId", orgUnitId)
        );

        jobRepository.save(job);

        // 2. 异步执行
        jobExecutor.executeAsync(job);

        // 3. 立即返回JobId
        return toBatchJobDTO(job);
    }

    /**
     * 查询Job状态
     */
    public BatchJobDTO getJobStatus(String jobId) {
        PlaceBatchJob job = jobRepository.findById(jobId).orElseThrow();
        return toBatchJobDTO(job);
    }

    /**
     * 重试失败项
     */
    public BatchJobDTO retryFailedItems(String jobId) {
        PlaceBatchJob job = jobRepository.findById(jobId).orElseThrow();

        if (job.getFailureCount() == 0) {
            throw new IllegalStateException("没有失败项");
        }

        // 创建新Job，只包含失败项
        PlaceBatchJob retryJob = job.createRetryJob();
        jobRepository.save(retryJob);

        jobExecutor.executeAsync(retryJob);

        return toBatchJobDTO(retryJob);
    }
}
```

#### Job执行器

```java
// application/place/PlaceBatchJobExecutor.java

@Component
@RequiredArgsConstructor
public class PlaceBatchJobExecutor {

    private final PlaceRepository placeRepository;
    private final TransactionTemplate transactionTemplate;

    /**
     * 异步执行Job
     */
    @Async("batchJobExecutor")
    public void executeAsync(PlaceBatchJob job) {
        try {
            job.start();
            jobRepository.save(job);

            List<Long> pendingIds = job.getPendingItemIds();

            // 分批处理（每批100个）
            int batchSize = 100;
            for (int i = 0; i < pendingIds.size(); i += batchSize) {
                List<Long> batch = pendingIds.subList(
                    i,
                    Math.min(i + batchSize, pendingIds.size())
                );

                processBatch(job, batch);

                // 更新进度
                job.updateProgress();
                jobRepository.save(job);
            }

            job.complete();
            jobRepository.save(job);

        } catch (Exception e) {
            job.fail(e.getMessage());
            jobRepository.save(job);
        }
    }

    /**
     * 处理单批（独立事务）
     */
    private void processBatch(PlaceBatchJob job, List<Long> placeIds) {
        String operation = job.getOperation();
        Map<String, Object> params = job.getParameters();

        if ("BATCH_ASSIGN_ORG".equals(operation)) {
            Long orgUnitId = (Long) params.get("orgUnitId");

            for (Long placeId : placeIds) {
                // 每个场所独立事务
                transactionTemplate.execute(status -> {
                    try {
                        UniversalPlace place = placeRepository.findById(placeId).orElseThrow();
                        place.setOrgUnitId(orgUnitId);
                        placeRepository.save(place);

                        job.markItemSuccess(placeId);

                    } catch (Exception e) {
                        job.markItemFailure(placeId, e.getMessage());
                    }
                    return null;
                });
            }
        }
    }
}
```

```sql
DELIMITER $$
CREATE PROCEDURE increment_occupancy_if_available(
  IN p_place_id BIGINT,
  OUT p_success BOOLEAN
)
BEGIN
  DECLARE affected INT;

  UPDATE places
  SET current_occupancy = current_occupancy + 1
  WHERE id = p_place_id
    AND current_occupancy < capacity;

  SELECT ROW_COUNT() INTO affected;
  SET p_success = (affected > 0);
END$$
DELIMITER ;
```

```java
// infrastructure/scheduler/CapacityAlertScheduler.java

@Component
@RequiredArgsConstructor
@Slf4j
public class CapacityAlertScheduler {

    private final PlaceRepository placeRepository;
    private final RedisTemplate<String, Object> redisTemplate;

    /**
     * 每5分钟计算一次容量预警
     */
    @Scheduled(cron = "0 */5 * * * *")
    public void updateCapacityAlerts() {
        log.info("开始计算容量预警...");

        // 1. 数据库层面计算（只查询高使用率）
        List<UniversalPlace> places = placeRepository.findHighOccupancyPlaces(0.8, 1000);

        // 2. 转换为预警对象
        List<CapacityAlert> alerts = places.stream()
            .map(this::toCapacityAlert)
            .sorted(Comparator.comparing(CapacityAlert::getOccupancyRate).reversed())
            .collect(Collectors.toList());

        // 3. 写入Redis
        String cacheKey = "capacity:alerts";
        redisTemplate.opsForValue().set(cacheKey, alerts, 10, TimeUnit.MINUTES);

        log.info("容量预警计算完成，共{}个预警", alerts.size());

        // 4. 发送通知（CRITICAL级别）
        sendCriticalAlerts(alerts);
    }

    private void sendCriticalAlerts(List<CapacityAlert> alerts) {
        List<CapacityAlert> criticalAlerts = alerts.stream()
            .filter(a -> "CRITICAL".equals(a.getLevel()))
            .collect(Collectors.toList());

        if (!criticalAlerts.isEmpty()) {
            // 发送邮件/短信/企业微信通知
            notificationService.sendAlert(
                "容量严重预警",
                "以下场所容量使用率超过95%：\n" + formatAlerts(criticalAlerts)
            );
        }
    }
}
```

### 改动5：并发控制（数据库原子操作）

```java
// application/place/PlaceOccupancyService.java

@Service
@Transactional
@RequiredArgsConstructor
public class PlaceOccupancyService {

    private final PlaceRepository placeRepository;
    private final PlaceOccupantRepository occupantRepository;

    /**
     * 入住（原子操作，防止超卖）
     */
    public void checkIn(Long placeId, Long userId) {
        // 1. 先尝试原子性增加占用数
        int updated = placeRepository.incrementOccupancyIfAvailable(placeId);

        if (updated == 0) {
            // 容量已满或场所不存在
            UniversalPlace place = placeRepository.findById(placeId).orElseThrow();
            throw new IllegalStateException(
                String.format("容量已满: %d/%d",
                             place.getCurrentOccupancy(),
                             place.getCapacity())
            );
        }

        // 2. 创建入住记录
        try {
            UniversalPlaceOccupant occupant = UniversalPlaceOccupant.create(
                placeId,
                userId,
                OccupantType.USER
            );
            occupantRepository.save(occupant);

        } catch (Exception e) {
            // 回滚占用数
            placeRepository.decrementOccupancyIfPossible(placeId);
            throw e;
        }
    }
}
```
